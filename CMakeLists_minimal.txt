cmake_minimum_required(VERSION 3.12)
project(PianoSynthModular VERSION 1.0.0 LANGUAGES CXX)

# Set C++ standard
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Build configuration
message(STATUS "=== Piano Synth Minimal Build Configuration ===")
message(STATUS "Build type: ${CMAKE_BUILD_TYPE}")

# Find required packages
find_package(GTest REQUIRED)

# Include directories
set(SHARED_INCLUDE_DIRS
    ${CMAKE_SOURCE_DIR}
    ${CMAKE_SOURCE_DIR}/shared
    ${CMAKE_SOURCE_DIR}/shared/interfaces
    ${CMAKE_SOURCE_DIR}/shared/utils
    ${CMAKE_SOURCE_DIR}/core
)

# Shared utilities library (minimal)
add_library(piano_shared_minimal STATIC
    shared/utils/json_config.cpp
)

target_include_directories(piano_shared_minimal PUBLIC ${SHARED_INCLUDE_DIRS})

# Tests for modular components only
enable_testing()

add_subdirectory(tests)

message(STATUS "==============================================")
